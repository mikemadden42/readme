========

softwareupdate --fetch-full-installer --full-installer-version 12.6.1

hdiutil create -o /tmp/macOS-12.6.1 -size 14000m -volname macOS-12.6.1 -layout SPUD -fs HFS+J

hdiutil attach /tmp/macOS-12.6.1.dmg -noverify -mountpoint /Volumes/macOS-12.6.1

sudo /Applications/Install\ macOS\ Monterey.app/Contents/Resources/createinstallmedia --volume /Volumes/macOS-12.6.1 --nointeraction

hdiutil detach /Volumes/Install*

# If you encounter an issue with this detach pass -force.
hdiutil detach /Volumes/Install* -force

hdiutil convert /tmp/macOS-12.6.1.dmg -format UDTO -o ~/Desktop/macOS-12.6.1.cdr

mv ~/Desktop/macOS-12.6.1.cdr ~/Desktop/macOS-12.6.1.iso
